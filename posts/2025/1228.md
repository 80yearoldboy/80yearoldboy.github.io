## B3759 [信息与未来2021]文本分类-题解
[题目传送门](https://www.luogu.com.cn/problem/B3759)

### 题意简述：
给定一句话，编写程序判断这句话是拼音还是英语
题解里大部分都是打表的做法，但是~~因为我太懒了~~秉承着“精益求精”的态度，我想到可不可以把声母和韵母分开来存，就可以~~更好的偷懒~~极大的节省了代码量。
于是就有了以下代码：
```c++
#include<bits/stdc++.h>
using namespace std;
string shenmu[24]={"","a","b","c","d","e","f","g","h","j","k","l","m","n","o","p","q","r","s","t","w","x","y","z"};//声母 ch,zh,sh特判
string yunmu[59]={"","i","n","ng","o","a","ai","an","ang","ao","ei","en","eng","ian","iao","ie","in","ing","u","e","ha","hai","han","hang","hao","he","hen","heng","hi","hong","hou","hu","huai","huan","huang","hui","hun","huo","ong","ou","uan","ui","un","uo","iu","r","ua","uai","uang","ia","iang","iong","ue","ü","üe","v","ve","hei","hua"};//韵母 ai,an,ang,ao以及ch,sh,zh开头的特判
int t,n;
string obj;
bool eng=false;
bool check(string x){
	bool flg=false;
	//找匹配的声母
	for(int i=1;i<=23;i++){
		if(x[0]==shenmu[i][0]){
			flg=true;//如果找到了匹配的声母则记录
		}
	}
	if(!flg)return false;//flg为假则直接判断为英语
	x.erase(0,1);//删去声母，则剩下的就是韵母
	//找匹配的韵母
	for(int i=1;i<=58;i++){
		if(x==yunmu[i]){
			return true;//只要找到了就为拼音
		}
	}
	return false;//否则为英语
}
int main(){
	//开始写之前先叠个甲
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	cin>>t;//读入数据组数
	while(t--){
		eng=false;//默认为拼音
		cin>>n;
		for(int i=1;i<=n;i++){
			cin>>obj;
			if(!check(obj))eng=true;//只要有一个词判断为英语，这句话就为英语
		}
		if(eng)cout<<"English\n";
		else cout<<"Pinyin\n";
	}
	return 0;
}
```

我纳闷了好久，然后我想到了a,e,o这三个拼音。
由于找不到匹配的韵母，所以会被认为是英语。
加了一个特判，果然对了。
### AC代码：
```c++
#include<bits/stdc++.h>
using namespace std;
string shenmu[24]={"","a","b","c","d","e","f","g","h","j","k","l","m","n","o","p","q","r","s","t","w","x","y","z"};//声母 ch,zh,sh特判
string yunmu[59]={"","i","n","ng","o","a","ai","an","ang","ao","ei","en","eng","ian","iao","ie","in","ing","u","e","ha","hai","han","hang","hao","he","hen","heng","hi","hong","hou","hu","huai","huan","huang","hui","hun","huo","ong","ou","uan","ui","un","uo","iu","r","ua","uai","uang","ia","iang","iong","ue","ü","üe","v","ve","hei","hua"};//韵母 ai,an,ang,ao以及ch,sh,zh开头的特判
int t,n;
string obj;
bool eng=false;
bool check(string x){
	if(x=="a"||x=="e"||x=="o")return true;//a,e,o这三个要特判,这很重要！！！
	bool flg=false;
	//找匹配的声母
	for(int i=1;i<=23;i++){
		if(x[0]==shenmu[i][0]){
			flg=true;//如果找到了匹配的声母则记录
		}
	}
	if(!flg)return false;//flg为假则直接判断为英语
	x.erase(0,1);//删去声母，则剩下的就是韵母
	//找匹配的韵母
	for(int i=1;i<=58;i++){
		if(x==yunmu[i]){
			return true;//只要找到了就为拼音
		}
	}
	return false;//否则为英语
}
int main(){
	//开始写之前先叠个甲
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	cin>>t;//读入数据组数
	while(t--){
		eng=false;//默认为拼音
		cin>>n;
		for(int i=1;i<=n;i++){
			cin>>obj;
			if(!check(obj))eng=true;//只要有一个词判断为英语，这句话就为英语
		}
		if(eng)cout<<"English\n";
		else cout<<"Pinyin\n";
	}
	return 0;//打完收工！
}
```
